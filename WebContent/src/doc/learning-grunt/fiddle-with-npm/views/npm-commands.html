npm commands

install a module

install a npm module

npm install <npm_module_name>

install and save module to dependencies list

npm install <npm_module_name> --save

install and save module to devDependencies list

npm install <npm_module_name> --save-dev

install and save module to optionalDependencies list

npm install <npm_module_name> --save-optional

uninstall a module

uninstall a npm module

npm uninstall <npm_module_name>

uninstall module from dependencies and update package.json

npm uninstall <npm_module_name> --save

uninstall module from devDependencies and update package.json

npm uninstall <npm_module_name> --save-dev

uninstall module from optionalDependencies and update package.json

npm uninstall <npm_module_name> --save-optional

list module

list module(s) installed locally

npm list;
npm ls;

list module(s) installed globally

npm list -g;
npm ls -g;

display node_moduels dir locally

display node_moduels dir locally

npm root;

search for a npm module

npm search <module_name>

npm search express;
