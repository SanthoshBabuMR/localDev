Understanding Jasmine

matchers

.toEqual

Compare if two expression(s) are equal, but doesn't care, if both expressions are of the same object.

Example: 

          
it("expect a to be 1", function(){
  var a = 1;
  expect(a).toEqual(1);
});

.toBe

Compare if two expression(s) are equal and of the same object. .toBe works similar to JavaScript's === operator.
The example given below, will produce the opposite result, when used with .toEqual matcher.
Check out: JavaScript Primitive Types vs Reference Types

Example: 

          
it("expect {} to not equal {}", function(){
  expect({}).not.toBe({});
});

.toBeTruthy

test if an expression evaluates to true.

Example: 

          
it("expect 1 to be true", function(){
  expect(1).toBeTruthy();	
});

.toBeFalsy

test if an expression evaluates to false.

Example: 

          
it("expect 0 to be false", function(){
  expect(0).toBeFalsy();	
});

.not

used to negate the result of jasmine matchers

Example: 

          
it("expect string '5' to not equal numeric 5", function() {
  expect('5').not.toEqual(5);
});

.toContain

Check if the expression contains a desired character or array item.

Example: 

          
it("expect var arr = [1,2,3] to contain 3", function() {
  var arr = [1,2,3];
  expect(arr).toContain(3);
});

.toBeDefined

test if a variable or function or expression is defined
Check out: Understanding JavaScript's 'undefined'

Example: 

          
it("expect fn to be defined", function() {
  function fn() {};
  expect(fn).toBeDefined();
});

.toBeUndefined

test if a variable or function or expression is undefined
Check out: Understanding JavaScript's 'undefined'

Example: 

          
it("expect var a to be defined", function() {
  var a;
  expect(a).toBeUndefined();
});

.toBeNull

test if a variable or expression evaluates to null

Example: 

          
it("expect null to be null", function() {
  expect(null).toBeNull();
});

.toBeNaN

test if a expression evaluates to NaN

Example: 

          
it("expect 5*'x' to be NaN", function() {
  expect(5*'x').toBeNaN();
});

.toBeGreaterThan

test if a expression is greater than another expression or value

Example: 

          
it("expects 'var b = [1,2,3]' to be greater than 'var a = [1,2]'", function(){
  var a = [1,2];
	var b = [1,2,3];
	expect(b).toBeGreaterThan(a);
});

.toBeLessThan

test if a expression is lesser than another expression or value

Example: 

          
it("expects literal 'a' to be less than literal 'z'", function(){
  expect('a').toBeLessThan('z');
});

.toBeCloseTo

allows you to check if a number is close to another number, given a certain amount of decimal precision as the second argument

Example: 

          
it("expect 48.542 to be close to 48.5", function(){
  expect(48.542).toBeCloseTo(48.5,1)
});

.toMatch

match an expression against regular expression
Check out: Regular Expressions

Example: 

          
it("expect myname@email.com to match \w+@\w+\.\w+", function(){
  expect("myname@email.com").toMatch(/\w+@\w+\.\w+/);
});

.toThrow

expect a function to throw an error

Example: 

          
it("expect function fn to throw an error", function() {
  var fn = function(x){
    if(x) {
      throw new Error();
    }			  
  };
  expect(fn.bind(null, true)).toThrow();
});
